```javascript
var numsTwoSum = [2, 7, 11, 15];
```

```javascript
//A-I

function twosum(numsTwoSum, targetTwoSum) {
    var hash = [];
    console.log(hash[0]); // undefined
   
    for(var i = 0; i < numsTwoSum.length; i++) {
        var num = numsTwoSum[i];
        if(hash[num] !== undefined) {
            return [hash[num], i]
        } else {
            hash[targetTwoSum - num] = i;
        }
    }
   
    return [];// not found
}
 
twosum(numsTwoSum, targetTwoSum);
```

```javascript
var twoSum2 = function(numsTwoSum, targetTwoSum) {
  var a = [];
  for (var i = 0; i < numsTwoSum.length; i++) {
    var tmp = targetTwoSum - numsTwoSum[i];
    if (a[tmp] !== undefined) return [a[tmp], i];
    a[numsTwoSum[i]] = i;
  }
  return [];// not found
};

twoSum2(numsTwoSum, targetTwoSum);
```
